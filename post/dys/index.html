using namespace std;
int n;
int heap[11]={0,1,8,2,11,12,4,6};
int ans[8]; 
void d(int i)
{
    int t;
    while(i*2<=n)
    {
        t=i;
        if(heap[2*i]<heap[i])
        {
            t=2*i;
        }
        else
        {
            t=i;
        }
        if(2*i+1<=n)
        {
            if(heap[2*i+1]<heap[t])
            {
                t=2*i+1;
            }
        }
        if(t!=i)
        {
            swap(heap[t],heap[i]);
            i=t;
        }
        else
            break;
    }
    return;
}

{
    heap[1]=10;
    n=7;

    int k = 0;

		k++;
		ans[k] = heap[1];
		heap[1] = heap[n];
		n--;
		d(1);
	}
	for(int i=1;i<=7;i++)
	{
		cout<<ans[i]<<" ";
	}
    return 0;
}
