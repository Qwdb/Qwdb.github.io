3. AC代码

#include<bits/stdc++.h>
using namespace std;
int e[501][501], book[501], s[501], top = 0, n, f,d, flag, ans,z;
void dfs(int u)
{
    if (book[u] == 1)
        return;
    book[u] = 1;
    top++;
    s[top] = u;
    if (u == z)
    {
        int minv = 999999999;
        for (int i = 1; i <= top - 1; i++)
        {
            //cout << s[i] << " ";
            if (minv > e[s[i]][s[i + 1]])
                minv = e[s[i]][s[i + 1]];
        }
        if (minv != 999999999)
        {
            flag = 1;
            for (int i = 1; i <= top - 1; i++)
            {
                e[s[i]][s[i + 1]] -= minv;
                e[s[i + 1]][s[i]] += minv;//在反向边上加上减去的流量 
            }
            ans += minv;//累加所有的流量 
        }
        top--;
        return;
    }
    for (int i = 1; i <= z; i++)
    {
        if (e[u][i] > 0)
        {
            dfs(i);
        }
    }
    top--;
    return;
}
