①最短路
Floyd 算法

for(int k=1; k<=n; k++)
{         
   for(int i=1; i<=n; i++)
   {
        for(int j=1; j<=n; j++)
        {
            if(MAP[i][k]+MAP[k][j]<MAP[i][j])
            {
                MAP[i][j]=MAP[i][k]+MAP[k][j];
            }
        }
    }
}
